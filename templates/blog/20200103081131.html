請求書の作成 - PDF::API2で帳票作成

PDF::API2で請求書を作成してみました。見積書、申込書、レシート、サービス利用明細など、本格的な帳票の作成の基本的なひな型として利用できます。
  • 日本語対応
  • 太字対応
  • フォントサイズ対応
  • 下線対応
  • 表の表示
  • 表のカラーリング
  • 単価、数量、価格
  • 価格合計
  • 印鑑押印用の四角
  • 会社ロゴの挿入

PDF::API2で書いた請求書のソースコード

PDF::API2で書いた請求書のソースコードです。 会社ロゴを入れる場合は、同一ディレクトリに、会社ロゴlogo.pngを入れて、ソースコードの該当箇所のコメントアウトをはずしてください。 
use strict;
use warnings;
use utf8;
use FindBin;

use PDF::API2;

my $pdf = PDF::API2->new;

$pdf->mediabox('A4');

my @page_size_infos = $pdf->mediabox;
my $page_width = $page_size_infos[2];
my $page_height = $page_size_infos[3];

my $default_font_size = 10;

# ページ余白
my $page_top_margin = 50;
my $page_bottom_margin = 50;
my $page_left_margin = 50;
my $page_right_margin = 50;

# 描画開始x座標
my $render_start_x = $page_left_margin;

# 描画開始y座標
my $render_start_y = $page_height - 1 - $page_top_margin;

# 描画終了x座標
my $render_end_x = $page_width - 1 - $page_right_margin;

# 描画終了y座標
my $render_end_y = $page_bottom_margin;

# 日本語に対応したTrueTypeフォントの読み込み
my $true_type_font_file = "$FindBin::Bin/AozoraMinchoRegular.ttf";
my $font = $pdf->ttfont($true_type_font_file);
my $true_type_font_bold_file = "$FindBin::Bin/AozoraMincho-bold.ttf";
my $font_bold = $pdf->ttfont($true_type_font_bold_file);

my $cur_row = 0;

my $item_height = 15;
my $item_width = ($render_end_x - $render_start_x) / 100;
my $text_bottom_margin = 3;

my $receive_company_end_items_count = 55;
my $send_company_start_items_count = 66;

my $table_text_left_margin = 3;

my $page = $pdf->page;

my $gfx = $page->gfx;

my $text = $page->text;
$text->font($font, $default_font_size);
$text->strokecolor('#000');
$text->fillcolor('#000');

my $line_width_basic = 0.3;
my $line_width_bold = 2;


# 一番上飾り太線
$gfx->move($render_start_x, $render_start_y);
$gfx->line($render_end_x, $render_start_y);
$gfx->linewidth(10);
$gfx->stroke;
$cur_row += 2;

# 申し込み日付
$text->translate($render_end_x, $render_start_y - $item_height * $cur_row + $text_bottom_margin);
$text->text_right('2019年10月14日');
$gfx->move($render_start_x + $send_company_start_items_count * $item_width, $render_start_y - $item_height * $cur_row);
$gfx->line($render_end_x, $render_start_y - $item_height * $cur_row);
$gfx->linewidth($line_width_basic);
$gfx->stroke;
$cur_row += 1;

# ロゴの開始位置を保存
my $cur_logo_row = $cur_row - 0.5;

# 請求書タイトル
$text->translate($render_start_x, $render_start_y - $item_height * $cur_row + $text_bottom_margin);
$text->font($font_bold, 20);
$text->text('請求書');
$cur_row += 3;

# 右列の開始位置を保存
my $cur_right_row = $cur_row + 2;

# 納品先企業名
my $receive_company_name = 'XXXXX株式会社';
$text->translate($render_start_x, $render_start_y - $item_height * $cur_row + $text_bottom_margin);
$text->font($font, $default_font_size);
$text->text($receive_company_name);
$text->translate($render_start_x + $receive_company_end_items_count * $item_width , $render_start_y - $item_height * $cur_row + $text_bottom_margin);
$text->text_right('御中');
$gfx->move($render_start_x, $render_start_y - $item_height * $cur_row);
$gfx->line($render_start_x + $receive_company_end_items_count * $item_width, $render_start_y - $item_height * $cur_row);
$gfx->linewidth($line_width_basic);
$gfx->stroke;
$cur_row += 4;

# 件名
my $subject_label = '件　名:';
$text->translate($render_start_x, $render_start_y - $item_height * $cur_row + $text_bottom_margin);
$text->font($font, $default_font_size);
$text->text($subject_label);
my $subject = '2019年3月3日の商品売買契約';
$text->translate($render_start_x + 10 * $item_width, $render_start_y - $item_height * $cur_row + $text_bottom_margin);
$text->font($font, $default_font_size);
$text->text($subject);
$gfx->move($render_start_x, $render_start_y - $item_height * $cur_row);
$gfx->line($render_start_x + $receive_company_end_items_count * $item_width, $render_start_y - $item_height * $cur_row);
$gfx->linewidth($line_width_basic);
$gfx->stroke;
$cur_row += 1;

# 納品日
my $delivery_date_label = '納品日:';
$text->translate($render_start_x, $render_start_y - $item_height * $cur_row + $text_bottom_margin);
$text->font($font, $default_font_size);
$text->text($delivery_date_label);
my $delivery_date = '2019年3月30日';
$text->translate($render_start_x + 10 * $item_width, $render_start_y - $item_height * $cur_row + $text_bottom_margin);
$text->font($font, $default_font_size);
$text->text($delivery_date);
$gfx->move($render_start_x, $render_start_y - $item_height * $cur_row);
$gfx->line($render_start_x + $receive_company_end_items_count * $item_width, $render_start_y - $item_height * $cur_row);
$gfx->linewidth($line_width_basic);
$gfx->stroke;
$cur_row += 4;

# 請求メッセージ
$text->translate($render_start_x, $render_start_y - $item_height * $cur_row + $text_bottom_margin);
$text->font($font, $default_font_size);
$text->text('下記の通り、ご請求申し上げます。');
$cur_row += 2;

# 金額
my $price_total_top_label = '金額';
$text->translate($render_start_x, $render_start_y - $item_height * $cur_row + $text_bottom_margin);
$text->font($font, $default_font_size);
$text->text($price_total_top_label);
my $price_total_top = '￥30,000円';
$text->translate($render_start_x + 48 * $item_width, $render_start_y - $item_height * $cur_row + $text_bottom_margin);
$text->font($font, $default_font_size + 5);
$text->text_right($price_total_top);
my $price_total_top_zeikomi = '税込';
$text->translate($render_start_x + 50 * $item_width, $render_start_y - $item_height * $cur_row + $text_bottom_margin);
$text->font($font, $default_font_size);
$text->text($price_total_top_zeikomi);
$gfx->move($render_start_x, $render_start_y - $item_height * $cur_row);
$gfx->line($render_start_x + $receive_company_end_items_count * $item_width, $render_start_y - $item_height * $cur_row);
$gfx->move($render_start_x, $render_start_y - $item_height * $cur_row - 2);
$gfx->line($render_start_x + $receive_company_end_items_count * $item_width, $render_start_y - $item_height * $cur_row - 2);
$gfx->linewidth($line_width_basic);
$gfx->stroke;
$cur_row += 2;

#=pod
# 会社ロゴ挿入
my $logo_image_file = "$FindBin::Bin/logo.png";
my $logo_image_object = $pdf->image_png($logo_image_file);
my $logo_image_width = 60;
$gfx->image(
  $logo_image_object,
  $render_start_x + $send_company_start_items_count * $item_width,
  $render_start_y - $item_height * $cur_logo_row - $logo_image_width,
  $logo_image_width,
  $logo_image_width,
);
#=cut

# 請求者
my $sender_company_name = '木本システム株式会社';
$text->translate($render_start_x + $send_company_start_items_count * $item_width, $render_start_y - $item_height * $cur_right_row + $text_bottom_margin);
$text->font($font, $default_font_size);
$text->text($sender_company_name);
$cur_right_row += 1;

# 郵便番号
my $sender_zip_code = '住所:123-4567';
$text->translate($render_start_x + $send_company_start_items_count * $item_width, $render_start_y - $item_height * $cur_right_row + $text_bottom_margin);
$text->font($font, $default_font_size);
$text->text($sender_zip_code);
$cur_right_row += 1;

# 住所
my $sender_address = '東京都港区〇〇 １２３－４';
$text->translate($render_start_x + $send_company_start_items_count * $item_width, $render_start_y - $item_height * $cur_right_row + $text_bottom_margin);
$text->font($font, $default_font_size);
$text->text($sender_address);
$cur_right_row += 1;

# 電話番号
my $sender_tel = 'TEL:090-1234-5678';
$text->translate($render_start_x + $send_company_start_items_count * $item_width, $render_start_y - $item_height * $cur_right_row + $text_bottom_margin);
$text->font($font, $default_font_size);
$text->text($sender_tel);
$cur_right_row += 1;

# FAX
my $sender_fax = 'FAX:090-1234-5679';
$text->translate($render_start_x + $send_company_start_items_count * $item_width, $render_start_y - $item_height * $cur_right_row + $text_bottom_margin);
$text->font($font, $default_font_size);
$text->text($sender_fax);
$cur_right_row += 1;

# 担当
my $sender_staff = '担当:田中太郎';
$text->translate($render_start_x + $send_company_start_items_count * $item_width, $render_start_y - $item_height * $cur_right_row + $text_bottom_margin);
$text->font($font, $default_font_size);
$text->text($sender_staff);

# 印鑑押印四角形
my $sleal_width = 60;
$gfx->rectxy(
  $render_start_x + $send_company_start_items_count * $item_width + $sleal_width * 2,
  $render_start_y - $item_height * ($cur_right_row - 1) + $sleal_width,
  $render_start_x + $send_company_start_items_count * $item_width + $sleal_width * 3,
  $render_start_y - $item_height * ($cur_right_row - 1)
);
$gfx->rectxy(
  $render_start_x + $send_company_start_items_count * $item_width,
  $render_start_y - $item_height * $cur_right_row ,
  $render_start_x + $send_company_start_items_count * $item_width + $sleal_width * 3,
  $render_start_y - $item_height * $cur_right_row - $sleal_width 
);
$gfx->poly(
  $render_start_x + $send_company_start_items_count * $item_width + $sleal_width,
  $render_start_y - $item_height * $cur_right_row ,
  $render_start_x + $send_company_start_items_count * $item_width + $sleal_width,
  $render_start_y - $item_height * $cur_right_row - $sleal_width 
);
$gfx->poly(
  $render_start_x + $send_company_start_items_count * $item_width + $sleal_width * 2,
  $render_start_y - $item_height * $cur_right_row ,
  $render_start_x + $send_company_start_items_count * $item_width + $sleal_width * 2,
  $render_start_y - $item_height * $cur_right_row - $sleal_width 
);
$gfx->strokecolor('#bbb');
$gfx->stroke;

# 見積書見出し背景
$gfx->rectxy($render_start_x, $render_start_y - $item_height * $cur_row, $render_end_x, $render_start_y - $item_height * ($cur_row + 1));
$gfx->fillcolor('#eee');
$gfx->fill;

# 見積書見出し上飾り太線
$gfx->poly($render_start_x, $render_start_y - $item_height * $cur_row, $render_end_x, $render_start_y - $item_height * $cur_row);
$gfx->linewidth($line_width_bold);
$gfx->strokecolor('#000');
$gfx->stroke;
$cur_row += 1;

# 見積書見出し下飾り太線
$gfx->poly($render_start_x, $render_start_y - $item_height * $cur_row, $render_end_x, $render_start_y - $item_height * $cur_row);
$gfx->linewidth($line_width_bold);
$gfx->stroke;

my $cur_column_items_count = 0;

# No
my $no_items_count = 10;
$text->translate($render_start_x + $cur_column_items_count * $item_width + $table_text_left_margin, $render_start_y - $item_height * $cur_row + $text_bottom_margin);
$text->font($font, $default_font_size);
$text->text('No.');
$cur_column_items_count += $no_items_count;

# 項目
my $name_items_count = 34;
$text->translate($render_start_x + $cur_column_items_count * $item_width + ($name_items_count * $item_width / 2), $render_start_y - $item_height * $cur_row + $text_bottom_margin);
$text->font($font, $default_font_size);
$text->text_center('項目');
$cur_column_items_count += $name_items_count;

# 数量
my $count_items_count = 14;
$text->translate($render_start_x + $cur_column_items_count * $item_width + ($count_items_count * $item_width / 2), $render_start_y - $item_height * $cur_row + $text_bottom_margin);
$text->font($font, $default_font_size);
$text->text_center('数量');
$cur_column_items_count += $count_items_count;

# 単価
my $unit_price_items_count = 14;
$text->translate($render_start_x + $cur_column_items_count * $item_width + ($unit_price_items_count * $item_width / 2), $render_start_y - $item_height * $cur_row + $text_bottom_margin);
$text->font($font, $default_font_size);
$text->text_center('単価');
$cur_column_items_count += $unit_price_items_count;

# 金額
my $price_items_count = 28;
$text->translate($render_start_x + $cur_column_items_count * $item_width + ($price_items_count * $item_width / 2), $render_start_y - $item_height * $cur_row + $text_bottom_margin);
$text->font($font, $default_font_size);
$text->text_center('金額');
